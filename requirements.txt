# ── CORE FRAMEWORK ──────────────────────────────────────────
Flask==3.1.3                  # Web framework — HTTP server, routing, JSON responses
Flask-Cors==6.0.0             # Cross-Origin Resource Sharing headers for frontend-backend comm

# ── FLASK DEPENDENCIES (installed automatically with Flask) ──
Werkzeug==3.1.7               # WSGI toolkit — request/response handling, routing
Jinja2==3.1.6                 # Template engine (used internally by Flask)
MarkupSafe==3.0.3             # Safe string handling for Jinja2
click==8.1.8                  # CLI utilities used by Flask dev server
itsdangerous==2.2.0           # Secure data signing (used by Flask sessions)
blinker==1.9.0                # Signal support for Flask extensions

# ── STANDARD LIBRARY (built-in, no install needed) ──────────
# sqlite3     — SQLite database engine (built into Python)
# hashlib     — SHA-256 password hashing
# secrets     — Cryptographically secure session token generation
# json        — JSON serialization/deserialization
# os          — File path and environment variable access

# ============================================================
# QUICK INSTALL GUIDE
# ============================================================
#
# 1. Make sure Python 3.10 or higher is installed:
#      python --version
#
# 2. (Optional) Create a virtual environment:
#      python -m venv venv
#      source venv/bin/activate       # Mac/Linux
#      venv\Scripts\activate          # Windows
#
# 3. Install dependencies:
#      pip install -r requirements.txt
#
# 4. Run the backend server:
#      cd backend
#      python server.py
#
# 5. Open the frontend:
#      Open frontend/index.html in any modern browser
#
# The backend runs at: http://localhost:5000
# No database setup needed — SQLite file created automatically
# ============================================================
